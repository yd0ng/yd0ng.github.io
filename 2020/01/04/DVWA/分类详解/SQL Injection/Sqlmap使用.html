sqlmap介绍

常用参数

--batch                #默认为yes
--dbs                #枚举数据库
--current-db        #指定当前数据库
--users                #枚举数据库用户
-D 库名               #指定数据库
--tables             #枚举表明
-T 表名               #制定表
--columns            #枚举列名
—C 列名               #指定列名
--dump(-all)        #获取数据
--level               #执行测试的等级（1~5）
--risk                #执行测试的风险（1~3）
--string            #匹配指定字符串
